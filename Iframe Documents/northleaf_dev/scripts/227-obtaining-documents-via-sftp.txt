<?php 

use GuzzleHttp\Client;
use GuzzleHttp\Psr7\Request;
class localApi 
{
    private function getHeaders() 
    {
        return [
                'Authorization' => 'Bearer ' . getenv('API_TOKEN'),
                'Content-Type' => 'application/json',
                'Accept' => 'application/json',
            ];
    }
    
    private function getClient() 
    {
        return  new Client([
                'curl' => [CURLOPT_SSL_VERIFYPEER => 0, CURLOPT_SSL_VERIFYHOST => 0],
                'allow_redirects' => false,
                'cookies' => true,
                'verify' => false
            ]);
    }

    public static function request($method, $url, $params = [], $data = null)
    {
        $queryString = http_build_query($params);
        $url = getenv('API_HOST') . "/{$url}?{$queryString}";
        // $url = getenv('API_HOST') . "/{$url}";

        if ($data) {
            $request = new Request($method, $url, self::getHeaders(), json_encode($data));
            
        } else {
            $request = new Request($method, $url, self::getHeaders());
        }
        
        try {
            $response = self::getClient()->send($request);
            $content = $response->getBody()->getContents();
            return json_decode($content, true);
        } catch (Exception $e) {
            print_r([
                'url' => $url,
                'error' => $e->getMessage(),
            ]);
            throw $e;
        }
    }

    public static function getApi($url, $params = []) {
        $response = self::request('GET', $url, $params);
        return $response;
    }

    public static function deleteApi($url, $params = []) {
        $response = self::request('DELETE', $url, $params);
        return $response;
    }

    public static function postApi($url, $data) {
        $response = self::request('POST', $url, [], $data);
        return $response;
    }

    public static function putApi($url, $data) {
        $response = self::request('PUT', $url, [], $data);
        return $response;
    }

    public static function uploadFileToRequest($requestId, $fileName) {
        try {
            $client = new Client(['verify' => false]);

            $headers = [
                'Authorization' => 'Bearer ' . getenv('API_TOKEN'),
                'Accept' => 'application/json',
            ];
            
            $request = $client->request('POST', getenv('API_HOST') . '/requests/' . $requestId . '/files', [
                'headers' => $headers,
                'multipart' => [
                    [
                        'name' => 'file',
                        'contents' => fopen($fileName, 'r')
                    ],
                ]
            ]);
            $fileUploaded = json_decode($request->getBody()->getContents());
            return $fileUploaded->fileUploadId;
        } catch (Exception $exception) {
            throw $exception;
        }
    }
}

function saveDocument($requestId, $documentName, $fileContent) {
    try {
        $search = array('-', ':', ',', ' ');
        $nameDocument = date('mdYHis') . '_' . $documentName;
        $fileName = str_replace($search, '_', $nameDocument);
        $FilePath  = '/tmp/' . $fileName;
        file_put_contents($FilePath, $fileContent);

        ////---- Upload Document Request Parent
        $uploadDocuId = localApi::uploadFileToRequest($requestId, $FilePath);
        $docInfo = array(
            'DOC_ID' => $uploadDocuId,
            'DOC_NAME' => $fileName
        );
        return $docInfo;
    } catch (Exception $exception) {
        throw $exception;
    }
}
