RUN apt-get update && apt-get install -y \
    git \
    curl \
    wget \
    jq \
    tree \
    ca-certificates \
    software-properties-common \
    build-essential \
    libmagick++-dev \
    libpng-dev \
    libjpeg62-turbo-dev \
    libfreetype6-dev \
    pdftk \
    --no-install-recommends \
    && rm -rf /var/lib/apt/lists/*


# =============================
# Instalar GD (PHP 7.3)
# =============================
RUN docker-php-ext-configure gd \
    --with-freetype-dir=/usr/include/ \
    --with-jpeg-dir=/usr/include/ \
    && docker-php-ext-install gd


# =============================
# Instalar Imagick
# =============================
RUN git clone https://github.com/mkoppanen/imagick.git /tmp/imagick && \
    cd /tmp/imagick && \
    phpize && \
    ./configure && \
    make && make install && \
    docker-php-ext-enable imagick && \
    rm -rf /tmp/imagick
